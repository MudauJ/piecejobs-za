Add a full payment and escrow system to PieceJobs ZA.

## PHASE 1 — DATABASE CHANGES
Run this SQL in Supabase:

CREATE TABLE IF NOT EXISTS payments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id UUID REFERENCES jobs(id),
  homeowner_email TEXT,
  worker_id UUID REFERENCES workers(id),
  amount INTEGER NOT NULL,
  platform_fee INTEGER NOT NULL,
  worker_payout INTEGER NOT NULL,
  payout_method TEXT DEFAULT 'bank',
  status TEXT DEFAULT 'pending',
  payfast_payment_id TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE payments DISABLE ROW LEVEL SECURITY;
GRANT ALL ON payments TO anon, authenticated;

ALTER TABLE workers ADD COLUMN IF NOT EXISTS payout_method TEXT DEFAULT 'bank';
ALTER TABLE workers ADD COLUMN IF NOT EXISTS bank_name TEXT;
ALTER TABLE workers ADD COLUMN IF NOT EXISTS bank_account TEXT;
ALTER TABLE workers ADD COLUMN IF NOT EXISTS flash_phone TEXT;
